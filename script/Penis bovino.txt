async function enviarScript(scriptText) {
	const lines = scriptText
		.split(/[\n\t]+/)
		.map(line => line.trim())
		.filter(line => line);
		
	const main = document.querySelector("#main");
	const textarea = main.querySelector(`div[contenteditable="true"]`);
	
	if (!textarea) throw new Error("Não há uma conversa aberta");
	
	for (const line of lines) {
		console.log(line);
		
		textarea.focus();
		document.execCommand('insertText', false, line);
		textarea.dispatchEvent(new Event('input', { bubbles: true })); // Melhor usar 'input' do que 'change'
		
		// Espera um pouco antes de enviar
		await new Promise(resolve => setTimeout(resolve, 100)); 
		
		// Simula o clique no botão de envio
		const sendButton = main.querySelector(`[data-testid="send"]`) || main.querySelector(`[data-icon="send"]`);
		if (sendButton) {
			sendButton.click();
		} else {
			throw new Error("Botão de envio não encontrado");
		}

		// Se não for a última linha, espera antes de enviar a próxima mensagem
		if (lines.indexOf(line) !== lines.length - 1) {
			await new Promise(resolve => setTimeout(resolve, 250));
		}
	}
	
	return lines.length;
}
